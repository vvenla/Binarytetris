1   package tetris.binarytetris.gui;
2   
3   import java.util.Scanner;
4   import tetris.binarytetris.logic.PalikkaTaulukko;
5   import tetris.binarytetris.logic.TetrisPeli;
6   
7   /**
8    *
9    * @author Venla Viljamaa
10   */
11  public class Tekstikayttis {
12  
13      private final TetrisPeli peli;
14      private final PalikkaTaulukko taulukko;
15      private final Scanner lukija;
16  //    private final int summa;
17      private final int leveys;
18  
19      public Tekstikayttis(int korkeus, int leveys, Scanner lukija, int summa) {
20          this.peli = new TetrisPeli(korkeus, leveys, summa);
21          this.taulukko = peli.getTaulukko();
22          this.lukija = lukija;
23  //        this.summa = summa;
24          this.leveys = leveys;
25      }
26  
27      public void pelaa() {
28          while (peli.onkoTaulukossaTilaa()) {
29              peli.setUusiArvo();
30              System.out.println("Seuraava palikka: " + peli.getUusiArvo());
31              paivitaPeli();
32          }
33          System.out.println("Peli loppui");
34      }
35  
36      public void paivitaPeli() {
37          System.out.print("Anna sarake: ");
38          String komento = lukija.nextLine();
39  
40          if (!komento.matches("[1-" + leveys + "]")) {
41              System.out.println("Väärä komento");
42          } else {
43              peli.paivita(Integer.parseInt(komento));
44          }
45          tulosta();
46      }
47  
48      public void tulosta() {
49          System.out.println();
50          for (int y = 0; y < taulukko.getKorkeus(); y++) {
51              for (int x = 0; x < taulukko.getLeveys(); x++) {
52                  System.out.print(taulukko.getPalikka(y, x).getArvo() + " ");
53              }
54              System.out.println();
55          }
56          System.out.println();
57      }
58  
59  }
