1   package tetris.binarytetris.ui;
2   
3   import java.util.Scanner;
4   import tetris.binarytetris.logic.PalikkaTaulukko;
5   import tetris.binarytetris.logic.PalikkaPeli;
6   
7   /**
8    * Tekstikäyttöliittymä testailua varten, ennen graafista käyttöliittymää.
9    *
10   * @author Venla Viljamaa
11   */
12  public class Tekstikayttis {
13  
14      private final PalikkaPeli peli;
15      private final PalikkaTaulukko taulukko;
16      private final Scanner lukija;
17      private final int leveys;
18  
19      /**
20       * Konstruktori.
21       *
22       * @param korkeus taulukon korkeus
23       * @param leveys taulukon leveys
24       * @param lukija Scanner
25       * @param summa haluttu summa
26       * @param loppusumma voittosumma
27       *
28       */
29      public Tekstikayttis(int korkeus, int leveys, Scanner lukija, int summa, int loppusumma) {
30          this.peli = new PalikkaPeli(korkeus, leveys, summa, loppusumma);
31          this.taulukko = peli.getTaulukko();
32          this.lukija = lukija;
33  //        this.summa = summa;
34          this.leveys = leveys;
35      }
36  
37      /**
38       * Metodi käynnistää pelin.
39       *
40       */
41      public void pelaa() {
42          while (!peli.peliHavitty()) {
43              peli.setUusiArvo();
44              System.out.println("Seuraava palikka: " + peli.getUusiArvo());
45              paivitaPeli();
46          }
47          System.out.println("Peli loppui");
48      }
49  
50      /**
51       * Metodi päivittää peliä.
52       *
53       */
54      public void paivitaPeli() {
55          System.out.print("Anna sarake: ");
56          String komento = lukija.nextLine();
57  
58          if (!komento.matches("[1-" + leveys + "]")) {
59              System.out.println("Väärä komento");
60          } else {
61              peli.paivita(Integer.parseInt(komento));
62          }
63          tulosta();
64      }
65  
66      /**
67       * Metodi tulostaa pelitilanteen.
68       *
69       */
70      public void tulosta() {
71          System.out.println();
72          for (int y = 0; y < taulukko.getKorkeus(); y++) {
73              for (int x = 0; x < taulukko.getLeveys(); x++) {
74                  System.out.print(taulukko.getPalikka(y, x).getArvo() + " ");
75              }
76              System.out.println();
77          }
78          System.out.println();
79      }
80  
81  }
